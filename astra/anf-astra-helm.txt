### Add helm repo of Helm Chart
helm repo add bitnami https://charts.bitnami.com/bitnami

### Install MariaDB only
helm install astramaria bitnami/mariadb

### Install MySQL only
helm install astramy bitnami/mysql

### Install PostgreSQL only
helm install astrapost bitnami/mysql

### Install WordPress
helm install astrawp \
  --set wordpressUsername=wpadmin \
  --set wordpressPassword=SXCN2bj3TNcA \
  --set mariadb.auth.rootPassword=1xg2KnICsM1t \
    bitnami/wordpress
or 
helm install astrawp bitnami/wordpress
#helm install astraakswp bitnami/wordpress persistence.storageClass=netapp-anf-perf-standard

### Uninstall 
helm delete astrawp
